DROP KEYSPACE dp;

CREATE KEYSPACE dp WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };

USE dp;


CREATE TABLE response (
	id   blob, 
	original_url     text,    
	url             text,    
	type            text,    
	provider_name    text,    
	provider_url     text,    
	provider_display text,    
	favicon_url      text,    
	title           text,    
	description     text,    
	date            timestamp,    
	published       varint,    
	lead            text,    
	content         text,
PRIMARY KEY (id, date)
);
 
 
CREATE TABLE author (
	id   blob, 
	name text, 
	url  text, 
PRIMARY KEY (url, id)
);

CREATE TABLE keyword (
	id    blob, 
	score int,    
	name  text, 
PRIMARY KEY (name, id)
);

CREATE TABLE entity (
	id    blob, 
	count int,    
	name  text, 
PRIMARY KEY (name, id)
);

CREATE TABLE image (
	id      blob,  
	caption text,  
	url     text,  
	width   int,     
	height  int,     
	entropy float, 
	size    int,   
PRIMARY KEY (url, id)
);

CREATE TABLE related (
	id              blob,  
	description     text,  
	title           text,  
	url             text,  
	thumbnail_width  int,     
	score           float, 
	thumbnail_height int,     
	thumbnail_url   text,
PRIMARY KEY (url, id)
);
