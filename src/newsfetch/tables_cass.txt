
DESCRIBE  

DROP KEYSPACE dp;
CREATE KEYSPACE dp WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };
USE dp;

CREATE TABLE author (
	id   blob, 
	date timestamp, 
	name text, 
	url  text, 
PRIMARY KEY (id, date)
);

CREATE TABLE entity (
	id    blob, 
	date timestamp, 
	count int,    
	name  text, 
PRIMARY KEY (id, date)
);

CREATE TABLE image (
	id      blob, 
	date	timestamp, 
	pic_index int, 
	caption text,  
	url     text,
	width   int,     
	height  int,     
	entropy float, 
	size    int,   
PRIMARY KEY (id, date)  
);

CREATE TABLE keyword (
	id    blob, 
	date	timestamp, 
	score int,    
	name  text, 
PRIMARY KEY (id, date)
);

CREATE TABLE related (
	id              blob, 
	date            timestamp,  
	description     text,  
	title           text,  
	url             text,  
	thumbnail_width  int,     
	score           float, 
	thumbnail_height int,     
	thumbnail_url   text,
PRIMARY KEY (id, date)  
);

CREATE TABLE response (
	id blob,
	original_url     text,    
	url             text,    
	type            text,    
	provider_name    text,    
	provider_url     text,    
	provider_display text,    
	favicon_url      text,    
	title           text,    
	description     text,    
	date            timestamp,    
	published       varint,    
	lead            text,    
	content         text,
PRIMARY KEY (id, date)
);
